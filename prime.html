def is_prime(n): if n <= 1: return False for i in range(2, int(n**0.5) + 1): if n % i == 0: return False return True def check_prime(event): # Get the input value from the webpage num_value = Element("num").element.value result_element = Element("result").element if num_value.isdigit(): num = int(num_value) if is_prime(num): result_element.innerText = f"{num} is a prime number." result_element.classList.add("show") else: result_element.innerText = f"{num} is not a prime number." result_element.classList.add("show") else: result_element.innerText = "Please enter a valid number." result_element.classList.remove("show") # Attach the Python function to the button click event Element("check-btn").element.onclick = check_prime
