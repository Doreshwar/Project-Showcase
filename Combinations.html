def factorial(num): if num == 0 or num == 1: return 1 else: return num * factorial(num - 1) def combinations(n, r): return factorial(n) // (factorial(r) * factorial(n - r)) def calculate_combinations(event): # Get the input values from the webpage n_value = Element("n").element.value r_value = Element("r").element.value result_element = Element("result").element if n_value.isdigit() and r_value.isdigit(): n = int(n_value) r = int(r_value) if r <= n: comb = combinations(n, r) result_element.innerText = f"C({n}, {r}) = {comb}" result_element.classList.add("show") # Add animation class else: result_element.innerText = "r must be less than or equal to n." result_element.classList.remove("show") # Hide animation for error message else: result_element.innerText = "Please enter valid non-negative integers." result_element.classList.remove("show") # Hide animation for error message # Attach the Python function to the button click event Element("calculate-btn").element.onclick = calculate_combinations
